AGENT INTEGRATION EXAMPLE
================================================================================

SCENARIO: Generate test cases for "User Authentication API"

from src.vector_db.search_engine import HybridSearchEngine
from your_llm_library import LLM  # OpenAI, Anthropic, etc.

# Initialize
search_engine = HybridSearchEngine()
llm = LLM(api_key="your-key")

# Step 1: Retrieve relevant documentation
results = search_engine.search(
    query="user authentication API endpoints workflow",
    k=10,
    search_mode='hybrid'
)

# Step 2: Extract context
context = "\n\n".join([
    f"[Source: {r.chunk_metadata.doc_name}, Page: {r.chunk_metadata.page_number}]\n"
    f"{r.chunk_metadata.text}"
    for r in results
])

# Step 3: Generate test cases
prompt = f"""
You are a QA engineer. Based on the following documentation:

{context}

Generate comprehensive test cases for the User Authentication API.
Include: positive tests, negative tests, edge cases, security tests.
Format as a structured list with test case ID, description, steps, expected result.
"""

test_cases = llm.generate(prompt)

# Step 4: Use test cases
print(test_cases)
